AuthGuard Component
// src/shared/components/AuthGuard.tsx import { Navigate, useLocation } from 'react-router-dom' import { useAuthStore } from '@/store/authStore' interface Props { children: React.ReactNode role?: 'admin' | 'pembuat_soal' | 'pengawas' } export function AuthGuard({ children, role }: Props) { const { user, roles } = useAuthStore() const location = useLocation() if (!user) { return <Navigate to="/login" state={{ from: location }} replace /> } if (role && !roles.includes(role)) { return <Navigate to="/dashboard" replace /> } return <>{children}</> }

Zustand Auth Store
// src/store/authStore.ts import { create } from 'zustand' import { persist, createJSONStorage } from 'zustand/middleware' interface User { id: string; name: string; email: string } interface AuthState { user: User | null schoolId: string | null roles: string[] setAuth: (u: User, schoolId: string, roles: string[]) => void clear: () => void hasRole: (role: string) => boolean } export const useAuthStore = create<AuthState>()( persist( (set, get) => ({ user: null, schoolId: null, roles: [], setAuth: (user, schoolId, roles) => set({ user, schoolId, roles }), clear: () => set({ user: null, schoolId: null, roles: [] }), hasRole: (role) => get().roles.includes(role), }), { name: 'auth', storage: createJSONStorage(() => sessionStorage), // tidak persist di localStorage partialize: (state) => ({ user: state.user, schoolId: state.schoolId, roles: state.roles }) } ) )
TanStack Query — API Layer Pattern
// src/features/sync/api/sync.api.ts import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query' import { apiClient } from '@/shared/utils/apiClient' import type { SyncJob, SyncJobRun, CreateSyncJobDto } from '@/shared/types' // Query keys — centralized untuk invalidation export const syncKeys = { all: ['sync-jobs'] as const, list: () => [...syncKeys.all, 'list'] as const, detail: (id: string) => [...syncKeys.all, id] as const, runs: (id: string) => [...syncKeys.all, id, 'runs'] as const, } export function useSyncJobs() { return useQuery({ queryKey: syncKeys.list(), queryFn: () => apiClient.get<SyncJob[]>('/api/v1/sync-jobs'), staleTime: 30_000, // 30 detik cache refetchInterval: 60_000, // auto-refresh tiap menit }) } export function useCreateSyncJob() { const qc = useQueryClient() return useMutation({ mutationFn: (dto: CreateSyncJobDto) => apiClient.post<SyncJob>('/api/v1/sync-jobs', dto), onSuccess: () => { qc.invalidateQueries({ queryKey: syncKeys.list() }) } }) } export function useSyncJobRuns(jobId: string) { return useQuery({ queryKey: syncKeys.runs(jobId), queryFn: () => apiClient.get<SyncJobRun[]>(`/api/v1/sync-jobs/${jobId}/runs`), refetchInterval: (data) => data?.some(r => r.status === 'running') ? 3000 : false, // poll jika ada yg running enabled: !!jobId, }) }
apiClient — Axios wrapper dengan interceptor
// src/shared/utils/apiClient.ts import axios from 'axios' import { useAuthStore } from '@/store/authStore' export const apiClient = axios.create({ baseURL: import.meta.env.VITE_API_URL, withCredentials: true, // send HttpOnly cookie headers: { 'X-Requested-With': 'XMLHttpRequest' } // CSRF hint }) // inject school_id ke setiap request apiClient.interceptors.request.use((config) => { const { schoolId } = useAuthStore.getState() if (schoolId) config.headers['X-School-ID'] = schoolId return config }) // handle 401 → redirect ke login apiClient.interceptors.response.use( (r) => r.data, async (err) => { if (err.response?.status === 401) { useAuthStore.getState().clear() window.location.href = '/login' } return Promise.reject(err) } )

// src/features/onboarding/pages/OnboardingWizard.tsx import { useReducer, useCallback } from 'react' import { useNavigate } from 'react-router-dom' import { StepSchoolInfo } from '../components/StepSchoolInfo' import { StepAdminAccount } from '../components/StepAdminAccount' import { StepPickIntegration } from '../components/StepPickIntegration' import { WizardProgress } from '../components/WizardProgress' import { useCreateSchool } from '../api/onboarding.api' type WizardState = { step: 1 | 2 | 3 schoolInfo?: { name: string; domain: string; phone: string } adminInfo?: { email: string; name: string } schoolId?: string } type Action = | { type: 'NEXT_FROM_1'; payload: WizardState['schoolInfo']; schoolId: string } | { type: 'NEXT_FROM_2'; payload: WizardState['adminInfo'] } | { type: 'BACK' } function reducer(s: WizardState, a: Action): WizardState { switch (a.type) { case 'NEXT_FROM_1': return { ...s, step: 2, schoolInfo: a.payload, schoolId: a.schoolId } case 'NEXT_FROM_2': return { ...s, step: 3, adminInfo: a.payload } case 'BACK': return { ...s, step: (Math.max(1, s.step - 1) as 1 | 2 | 3) } default: return s } } export default function OnboardingWizard() { const [state, dispatch] = useReducer(reducer, { step: 1 }) const navigate = useNavigate() const createSchool = useCreateSchool() const handleStep1 = useCallback(async (data) => { const { school_id } = await createSchool.mutateAsync(data) dispatch({ type: 'NEXT_FROM_1', payload: data, schoolId: school_id }) }, [createSchool]) const steps = ['Info Sekolah', 'Akun Admin', 'Integrasi'] return ( <div className="wizard-container"> <WizardProgress steps={steps} current={state.step} /> {state.step === 1 && <StepSchoolInfo onSubmit={handleStep1} />} {state.step === 2 && <StepAdminAccount schoolId={state.schoolId!} onSubmit={d => dispatch({ type: 'NEXT_FROM_2', payload: d })} onBack={() => dispatch({ type: 'BACK' })} />} {state.step === 3 && <StepPickIntegration schoolId={state.schoolId!} onDone={() => navigate('/dashboard')} onBack={() => dispatch({ type: 'BACK' })} />} </div> ) }

Security — Frontend Layer
CSP, cookie strategy, CORS, XSS mitigation, dan CSRF defense untuk SPA.
Checklist Keamanan Frontend
  • ✓
    HttpOnly + SameSite=Strict cookie
    JWT access token tidak pernah di localStorage. Cookie di-set oleh server, tidak bisa diakses JS.
  • ✓
    Content Security Policy (CSP) ketat
    default-src 'self'; script-src 'self' 'nonce-{nonce}'; blokir eval() dan inline script.
  • ✓
    CORS whitelist eksplisit
    API hanya terima Origin dari domain FE resmi. Bukan wildcard *.
  • ✓
    X-Requested-With header + SameSite=Strict
    Dua lapis CSRF defense tanpa perlu CSRF token terpisah.
  • ✓
    DOMPurify untuk HTML dari API
    Setiap content HTML dari server di-sanitize sebelum di-render. React escape otomatis untuk JSX.
  • ✓
    Dependency audit di CI
    npm audit + Snyk scan di setiap PR. Tidak boleh ada HIGH/CRITICAL vulnerability.
  • !
    Refresh token rotation
    Refresh token di Redis server-side. Silent refresh 1 menit sebelum expire. Revocation via Redis DEL.
  • i
    Subresource Integrity (SRI)
    Untuk external scripts dari CDN, tambahkan integrity hash. Vite bisa auto-generate via plugin.

# apps/web/Dockerfile # Stage 1: build FROM node:20-alpine AS builder WORKDIR /app # cache node_modules layer COPY package*.json ./ RUN npm ci --prefer-offline COPY . . ARG VITE_API_URL ARG VITE_AUTH_URL RUN npm run build # output ke /app/dist # Stage 2: serve — minimal nginx FROM nginx:1.25-alpine AS runtime COPY --from=builder /app/dist /usr/share/nginx/html COPY nginx.conf /etc/nginx/conf.d/default.conf # non-root user RUN chown -R nginx:nginx /usr/share/nginx/html USER nginx EXPOSE 80 CMD ["nginx", "-g", "daemon off;"] # Build: docker build --build-arg VITE_API_URL=https://api.sekolah.id -t school-web .
