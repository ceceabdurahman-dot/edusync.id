# apps/worker/src/mq/setup.py — declare topology saat startup import aio_pika async def setup_rabbitmq(connection: aio_pika.Connection): channel = await connection.channel() await channel.set_qos(prefetch_count=10) # 1 worker: 10 msg max in-flight # ── Dead-Letter Exchange (DLX) ────────────────── dlx = await channel.declare_exchange( "school.dlx", aio_pika.ExchangeType.TOPIC, durable=True, ) # ── Dead-Letter Queue ─────────────────────────── dlq = await channel.declare_queue( "sync.dlq", durable=True, arguments={"x-queue-type": "quorum"}, # quorum queue — HA ) await dlq.bind(dlx, routing_key="#") # tangkap semua DL messages # ── Main exchange ─────────────────────────────── main_exchange = await channel.declare_exchange( "school.sync", aio_pika.ExchangeType.TOPIC, durable=True, ) # ── Main sync queue (dengan DLX) ──────────────── sync_queue = await channel.declare_queue( "sync.jobs", durable=True, arguments={ "x-dead-letter-exchange": "school.dlx", # nack → DLQ otomatis "x-dead-letter-routing-key": "sync.failed", "x-queue-type": "quorum", # HA untuk prod "x-message-ttl": 3_600_000, # 1 jam max age }, ) # sync.{school_id}.{provider} → sync.jobs await sync_queue.bind(main_exchange, routing_key="sync.#") # ── Test connection queue ──────────────────────── test_queue = await channel.declare_queue("test.connection.jobs", durable=True) await test_queue.bind(main_exchange, routing_key="test.connection.#") # ── Report queue ──────────────────────────────── report_queue = await channel.declare_queue("report.jobs", durable=True) await report_queue.bind(main_exchange, routing_key="report.#") return { "exchange": main_exchange, "sync_queue": sync_queue, "dlq": dlq, } # Routing key convention: # sync.{school_id}.{provider} → untuk priority routing per tenant # sync.{school_id}.all → semua provider sekolah tsb # test.connection.{connection_id} # report.{school_id}.{report_type}

# src/services/mq_publisher.py import aio_pika, json from uuid import UUID class MQPublisher: def __init__(self, exchange: aio_pika.Exchange): self._exchange = exchange async def publish_sync_job( self, job_id: UUID, run_id: UUID, school_id: UUID, provider: str, payload: dict, ): routing_key = f"sync.{school_id}.{provider}" message = aio_pika.Message( body=json.dumps({ "job_id": str(job_id), "run_id": str(run_id), "school_id": str(school_id), "provider": provider, "payload": payload, }).encode(), delivery_mode=aio_pika.DeliveryMode.PERSISTENT, # survive broker restart content_type="application/json", message_id=str(run_id), # untuk deduplekasi headers={"school_id": str(school_id)}, ) await self._exchange.publish(message, routing_key=routing_key) return str(run_id)
Retry Strategy & DLQ Pattern
Exponential backoff dengan full jitter. Max 5 attempts. Circuit breaker untuk melindungi SIS vendor dari overload.
Backoff Formula
# apps/worker/src/retry/backoff.py import random, math def full_jitter(attempt: int, base: float = 2.0, cap: float = 60.0) -> float: """ Full Jitter — AWS-recommended untuk mencegah thundering herd. Formula: sleep = random_between(0, min(cap, base * 2^attempt)) Contoh output: attempt 1 → sleep ∈ [0, 4s] attempt 2 → sleep ∈ [0, 8s] attempt 3 → sleep ∈ [0, 16s] attempt 4 → sleep ∈ [0, 32s] attempt 5 → sleep ∈ [0, 60s] (capped) """ exp = min(cap, base * (2 ** attempt)) return random.uniform(0, exp) # Contoh: visualisasi distribusi delay # attempt=1: E[delay] ≈ 2s, max 4s # attempt=2: E[delay] ≈ 4s, max 8s # attempt=5: E[delay] ≈ 30s, max 60s # Total worst case: ~120s untuk 5 attempts

# apps/worker/src/dlq/handler.py import structlog from db.session import get_db from db.repositories.sync_job_repo import SyncJobRepo log = structlog.get_logger() async def send_to_dlq( job_id: str, school_id: str, run_id: str, error: str, attempts: int = 0, permanent: bool = False, payload: dict = None, ): async with get_db() as db: repo = SyncJobRepo(db) # Tulis ke DB DLQ await repo.create_dlq_entry( job_id=job_id, run_id=run_id, school_id=school_id, payload=payload or {}, error_reason=error, attempts=attempts, is_permanent=permanent, ) # Update run status ke failed await repo.fail_run(run_id, error) log.error("job.dlq", job_id=job_id, school_id=school_id, run_id=run_id, error=error, attempts=attempts, permanent=permanent, ) # Increment Prometheus counter from config.otel import dlq_total dlq_total.labels(reason=error.split(":")[0]).inc() async def reprocess_dlq_entry(dlq_id: str, school_id: str): """Re-enqueue job dari DLQ. Dipanggil dari API endpoint POST /dlq/{id}/reprocess.""" async with get_db() as db: repo = SyncJobRepo(db) entry = await repo.get_dlq_entry(dlq_id, school_id) if not entry: raise ValueError(f"DLQ entry {dlq_id} not found") if entry.is_permanent: raise ValueError("Cannot reprocess permanent failure — fix root cause first") # Buat run baru new_run_id = await repo.create_run( job_id=entry.job_id, school_id=entry.school_id, triggered_by="dlq_reprocess", ) await repo.mark_dlq_reprocessed(dlq_id, new_run_id) # Publish ke MQ from mq.publisher import get_publisher publisher = get_publisher() await publisher.publish_sync_job( job_id=entry.job_id, run_id=new_run_id, school_id=entry.school_id, provider=entry.payload.get("provider"), payload=entry.payload, ) log.info("dlq.reprocessed", dlq_id=dlq_id, new_run_id=new_run_id) return new_run_id

Worker Full Implementation
Consumer loop dengan error handling lengkap. Acknowledgment strategy: ack setelah DB commit, nack jika transient error.
# apps/worker/src/main.py — Consumer loop import asyncio, aio_pika, json, structlog from mq.setup import setup_rabbitmq from jobs.sync_job import run_sync_job from config.settings import settings log = structlog.get_logger() async def process_message(message: aio_pika.IncomingMessage): """Process 1 message. Ack/nack berdasarkan hasil.""" async with message.process(requeue=False): # auto-nack jika exception body = json.loads(message.body) job_id = body["job_id"] run_id = body["run_id"] school_id = body["school_id"] log.info("worker.received", job_id=job_id, run_id=run_id) try: await run_sync_job( ctx={}, job_id=job_id, school_id=school_id, run_id=run_id, ) # ack otomatis di akhir `async with` block jika tidak exception except Exception as e: log.error("worker.unhandled_error", error=str(e), job_id=job_id) # aio_pika auto-nack → message masuk DLX → DLQ raise async def main(): connection = await aio_pika.connect_robust( settings.RABBITMQ_URL, reconnect_interval=5, ) mq = await setup_rabbitmq(connection) log.info("worker.started", concurrency=settings.WORKER_CONCURRENCY) await mq["sync_queue"].consume( process_message, no_ack=False, # manual ack — penting untuk at-least-once ) await asyncio.Future() # run forever if __name__ == "__main__": asyncio.run(main())
Modul 5 — Security
Security Checklist (OWASP)
  • ✓
    Argon2id untuk password hashing
    memory=64MB, iterations=3, parallelism=4 — OWASP recommended. BUKAN bcrypt/MD5.
  • ✓
    JWT RS256 dengan verifikasi lengkap (iss/aud/exp/sub)
    Tidak pernah skip verifikasi. Algorithm list eksplisit — tidak boleh ["*"].
  • ✓
    HttpOnly + SameSite=Strict + Secure cookie
    JWT tidak pernah di localStorage. Access token cookie 15 menit, refresh 7 hari path-limited.
  • ✓
    CORS whitelist eksplisit (bukan wildcard *)
    Hanya domain FE resmi. Divalidasi di NGINX dan FastAPI middleware.
  • ✓
    Content Security Policy (CSP) ketat
    default-src 'self'; script-src 'nonce-based'; blokir eval() dan inline script.
  • ✓
    Parameterized queries — zero SQL injection
    SQLAlchemy ORM + explicit text() parameters. Tidak ada string concatenation SQL.
  • ✓
    Rate limiting berlapis (NGINX + FastAPI slowapi)
    Auth: 5/menit, API: 100/menit, Submit: 10/menit. Per IP per endpoint.
  • ✓
    Structured logging dengan redaksi sensitif
    Processor otomatis hapus: password, token, secret, credentials dari log.
  • ✓
    PostgreSQL RLS sebagai defense-in-depth
    Multi-layer isolasi tenant. RLS + app filter + JWT validation.
  • ✓
    Dependency scanning di CI (Trivy + npm audit)
    Block PR jika ada HIGH/CRITICAL CVE. Scan setiap push ke main.
  • !
    Rotasi secrets setiap 90 hari
    DB password, JWT signing key, API keys. Via K8s Secrets + rolling restart.
  • !
    Enkripsi kolom PII (pgcrypto) — opsional awal
    NIK, nomor HP, data kesehatan. Key di Vault. Direkomendasikan sebelum produksi penuh.
  • i
    Audit integrity (opsional: hash chaining)
    Setiap audit_event menyimpan hash dari event sebelumnya. Tamper-evident log chain.
Password Hashing — Argon2id
# src/services/auth_service.py — password hashing from argon2 import PasswordHasher from argon2.exceptions import VerifyMismatchError, VerificationError # Konfigurasi OWASP recommended (2024) ph = PasswordHasher( time_cost=3, # iterasi CPU memory_cost=65536, # 64 MB RAM — cegah ASIC/GPU attack parallelism=4, # 4 threads hash_len=32, salt_len=16, ) def hash_password(raw: str) -> str: return ph.hash(raw) def verify_password(hashed: str, raw: str) -> bool: try: ph.verify(hashed, raw) # Rehash jika parameter berubah (otomatis upgrade) if ph.check_needs_rehash(hashed): return hash_password(raw) # caller simpan hash baru ke DB return True except (VerifyMismatchError, VerificationError): return False # Contoh hash output: # $argon2id$v=19$m=65536,t=3,p=4$salt16bytes$hash32bytes # # Waktu hash: ~0.3-0.5 detik (intentional — lambat untuk attacker) # Cost di 100k login/hari: trival. Cost untuk attacker: very high.
JWT / OIDC Validation
# JWT validation — 6 hal yang WAJIB diverifikasi ################################################## # 1. Algorithm restriction — TIDAK pakai "none" atau ["*"] algorithms=["RS256"] # explicit. RS256 lebih aman dari HS256 untuk multi-tenant # 2. Signature verification — otomatis dari public key jwt.decode(token, public_key, algorithms=["RS256"]) # 3. Issuer check (iss) issuer = "https://auth.sekolah.id/realms/school" # harus match EXACTLY # 4. Audience check (aud) audience = "school-api" # token harus punya aud field ini # 5. Expiry check (exp) # otomatis via jwt library. Access token: 15 menit. # 6. Required claims options={"require": ["sub", "iss", "aud", "exp", "school_id"]} ################################################## # Refresh token strategy # Access token (AT): HttpOnly cookie, 15 menit, path="/" # Refresh token (RT): HttpOnly cookie, 7 hari, path="/api/v1/auth/refresh" # # RT disimpan di Redis untuk revocation: async def store_refresh_token(user_id: str, rt: str, ttl: int = 604800): # key: "rt:{sha256(rt)}" → user_id key = "rt:" + hashlib.sha256(rt.encode()).hexdigest() await redis.set(key, user_id, ex=ttl) async def blacklist_refresh_token(rt: str): key = "rt:" + hashlib.sha256(rt.encode()).hexdigest() await redis.delete(key) # revoke immediately # Verifikasi AT via JWKS endpoint untuk key rotation: # GET https://auth.sekolah.id/.well-known/jwks.json # Cache JWKS 1 jam di Redis. Refresh jika kid tidak ditemukan.

Secrets Management & Rotation
Rotation Checklist (90 hari)
  • 1
    Generate credential baru
    openssl rand -base64 32 untuk DB password, JWT secret. Catat di Vault.
  • 2
    Update K8s Secret / Vault
    kubectl create secret ... --dry-run=client | kubectl apply -f -
  • 3
    Rolling restart pod
    kubectl rollout restart deployment/api-server. Zero downtime via rolling update.
  • 4
    Revoke credential lama di DB
    ALTER ROLE app_user PASSWORD 'new_pass'; Setelah semua pod restart.
  • 5
    Verifikasi health check
    Pantau /health endpoint + error rate selama 10 menit setelah rotation.
PII Handling & Data Minimization
Prinsip Data Minimization
  • ✓
    Hanya kumpulkan data yang dibutuhkan
    Tidak menyimpan foto, video, atau data biometrik kecuali eksplisit dibutuhkan.
  • ✓
    Audit log bebas PII
    audit_event.metadata TIDAK boleh berisi nama lengkap, NIK, atau nomor HP. Gunakan ID saja.
  • ✓
    Log tidak mengandung PII
    structlog processor redact key: nama, nik, email, phone, password. Replace dengan [REDACTED].
  • !
    Enkripsi pgcrypto untuk kolom sensitif
    NIK, nomor HP siswa. Implementasikan sebelum go-live dengan data riil.
  • i
    Kebijakan retensi data
    Setelah school suspend 2 tahun: anonymize data siswa (replace dengan hash). Setelah 5 tahun: delete.

6 Langkah Awal Implementasi
Practical checklist untuk tim engineering. Selesaikan urut — setiap langkah memblokir langkah berikutnya.
1
Hari 1-2: Monorepo + Docker Compose Dev
Init Turborepo. Setup docker-compose.yml: PostgreSQL 15, Redis 7, RabbitMQ 3.12, pgAdmin. Buat .env.example. Jalankan docker compose up -d → semua service healthy. Setup scripts/db-init.sql untuk create role, database, extensions.
2
Hari 2-4: Database Schema + RLS + Migrations
Jalankan DDL core, sync, audit (DDL dari modul database). Setup Alembic. Apply RLS policies. Verifikasi: connect sebagai app_user, pastikan hanya bisa akses dengan school_id benar. Buat seed data untuk 2 sekolah uji. Test RLS isolation.
3
Hari 3-6: Auth OIDC + JWT + Cookie Strategy
Setup Keycloak di docker-compose. Buat realm + client. Implementasi FastAPI auth router (POST /oidc/callback, /refresh, /logout). Set HttpOnly cookie. Test: login → cookie ada → /me return user info → logout → cookie cleared. FE login page + redirect flow.
4
Minggu 2: Onboarding Flow FE + BE
POST /api/v1/schools endpoint + onboarding wizard steps. FE: OnboardingWizard component (Zustand + RHF + Zod). Test full flow: daftar sekolah → setup admin → pilih integrasi. Verifikasi: school.status = active setelah selesai.
5
Minggu 2-3: Sync Worker + RabbitMQ + Retry/DLQ
Setup aio_pika consumer. Buat 1 SIS adapter (mock atau vendor nyata). Implementasi run_sync_job() dengan circuit breaker + full jitter backoff. Test: trigger sync → lihat di sync_job_run → data masuk DB. Test failure: mock SIS error → DLQ populated.
6
Minggu 3-4: OTel + Prometheus + GitHub Actions CI/CD
Setup OTel collector di docker-compose. Instrument FastAPI + worker. Grafana dashboard minimal (error rate, p99, queue depth). GitHub Actions: lint → test → build → deploy staging. Buat restore drill script. Jalankan drill pertama. Document RTO aktual.
