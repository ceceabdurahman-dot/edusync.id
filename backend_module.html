Module Boundaries
Setiap modul punya tanggung jawab yang jelas. Tidak ada import silang antar domain service.
auth/
Verifikasi JWT dari OIDC provider. Extrak claims (user_id, school_id, roles). Tidak tahu tentang business logic apapun.
Depends on: Keycloak public key, redis (token blacklist). Used by: semua route via Depends()
schools/
CRUD school entity. Wizard onboarding (step 1–3). Update status (onboarding→active). Tidak tahu tentang integrasi atau sync.
Depends on: school_repo, audit_service. Called by: FE onboarding wizard.
integrations/
CRUD integration_connection. Manage field mapping (integration_field_map). Test koneksi (async via worker). Tidak eksekusi sync.
Depends on: integration_service, adapter_registry, MQ publisher. Called by: FE setup page.
sync/
Create dan trigger sync_job. Poll status. Fetch run logs. Tidak mengeksekusi sync sendiri — delegasi ke worker via RabbitMQ.
Depends on: sync_service, MQ publisher, sync_job_repo. Called by: FE dashboard, scheduler.
users/
CRUD user_account per tenant. Assign/revoke role. Suspend user. Invite via email. Scoped ke school_id dari JWT.
Depends on: user_repo, role_repo, email_service, audit_service.
admin/
Platform-level operations: isolasi tenant, retensi data, global metrics. Hanya bisa diakses role super_admin — bukan school admin biasa.
Depends on: semua repo. Required role: super_admin (dari JWT platform claim).
services/
Business logic layer. Tidak ada SQL langsung — delegasi ke repository. Tidak ada HTTP knowledge — dipanggil oleh router via Depends().
Depends on: repositories. Called by: routers.
integrations/adapters/
SIS connector implementations. Setiap adapter implement AbstractSISAdapter. Dipilih via registry berdasarkan provider string.
Depends on: httpx (REST), paramiko (SFTP), pandas (CSV). Called by: worker sync_job.

# src/middleware/auth_middleware.py from fastapi import Depends, HTTPException, Request, status from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials import jwt from .config.settings import settings security = HTTPBearer() async def verify_jwt( request: Request, creds: HTTPAuthorizationCredentials = Depends(security) ) -> dict: """Decode dan validasi JWT. Inject ke request.state.""" token = creds.credentials try: payload = jwt.decode( token, settings.OIDC_PUBLIC_KEY, algorithms=["RS256"], # WAJIB explicit — jangan ["*"] options={ "verify_iss": True, # cek issuer "verify_aud": True, # cek audience "verify_exp": True, # cek expiry "require": ["sub", "iss", "aud", "exp"], }, audience=settings.JWT_AUDIENCE, issuer=settings.OIDC_ISSUER, ) except jwt.ExpiredSignatureError: raise HTTPException(status.HTTP_401_UNAUTHORIZED, detail="Token expired") except jwt.InvalidTokenError as e: raise HTTPException(status.HTTP_401_UNAUTHORIZED, detail="Invalid token") request.state.user_id = payload["sub"] request.state.school_id = payload.get("school_id") request.state.roles = payload.get("roles", []) return payload # Dependency factory untuk role guard def require_role(*roles: str): async def _guard(payload: dict = Depends(verify_jwt)) -> dict: user_roles = set(payload.get("roles", [])) if not user_roles.intersection(roles): raise HTTPException(status.HTTP_403_FORBIDDEN, detail="Insufficient role") return payload return _guard # Helper untuk ekstrak school_id dari JWT def get_current_school_id(request: Request) -> str: school_id = request.state.school_id if not school_id: raise HTTPException(status.HTTP_400_BAD_REQUEST, detail="school_id missing") return school_id

# src/api/v1/auth/router.py from fastapi import APIRouter, Depends, Response, Request from fastapi.responses import JSONResponse from .schemas import OIDCCallbackRequest, TokenResponse, UserInfoResponse from services.auth_service import AuthService router = APIRouter(prefix="/auth", tags=["auth"]) @router.post("/oidc/callback", response_model=UserInfoResponse) async def oidc_callback( payload: OIDCCallbackRequest, response: Response, svc: AuthService = Depends() ): tokens = await svc.exchange_code(payload.code, payload.redirect_uri) # Set HttpOnly cookie — TIDAK boleh diakses JavaScript response.set_cookie( key="access_token", value=tokens.access_token, httponly=True, secure=True, samesite="strict", max_age=900, # 15 menit path="/", ) response.set_cookie( key="refresh_token", value=tokens.refresh_token, httponly=True, secure=True, samesite="strict", max_age=86400 * 7, # 7 hari path="/api/v1/auth/refresh", # path-limited ) return tokens.user_info @router.post("/logout") async def logout(request: Request, response: Response, svc: AuthService = Depends()): refresh = request.cookies.get("refresh_token") if refresh: await svc.blacklist_token(refresh) response.delete_cookie("access_token") response.delete_cookie("refresh_token") return {"ok": True}

Worker Implementation
ARQ + RabbitMQ. Full retry dengan exponential backoff + jitter. Circuit breaker pattern.
jobs/sync_job.py — Complete
# apps/worker/src/jobs/sync_job.py import asyncio, random, math from uuid import UUID import structlog from opentelemetry import trace from db.session import get_db from db.repositories.sync_job_repo import SyncJobRepo from integrations.adapters.registry import get_adapter from dlq.handler import send_to_dlq from retry.circuit_breaker import CircuitBreaker, CircuitOpen log = structlog.get_logger() tracer = trace.get_tracer("school.worker") MAX_ATTEMPTS = 5 BASE_DELAY_S = 2.0 class TransientError(Exception): pass class PermanentError(Exception): pass def full_jitter(attempt: int, base: float = BASE_DELAY_S, cap: float = 60.0) -> float: """Full jitter — mencegah thundering herd.""" exp = min(cap, base * (2 ** attempt)) return random.uniform(0, exp) async def run_sync_job(ctx: dict, job_id: str, school_id: str, run_id: str): with tracer.start_as_current_span("sync_job.run") as span: span.set_attribute("job.id", job_id) span.set_attribute("school.id", school_id) bound_log = log.bind(job_id=job_id, school_id=school_id, run_id=run_id) bound_log.info("sync.started") async with get_db() as db: repo = SyncJobRepo(db) await repo.set_run_status(run_id, "running") # Ambil adapter berdasarkan provider adapter = get_adapter(job_id=job_id) cb = CircuitBreaker(key=f"sis:{school_id}") for attempt in range(1, MAX_ATTEMPTS + 1): try: # Circuit breaker akan raise CircuitOpen jika OPEN async with cb: data = await adapter.fetch_data() # Upsert ke DB (idempotent) async with get_db() as db: repo = SyncJobRepo(db) result = await repo.upsert_records(school_id, data) await repo.complete_run( run_id, records_synced=result.inserted + result.updated, records_failed=result.failed, ) bound_log.info("sync.completed", attempt=attempt, records_synced=result.inserted + result.updated ) return # ─ success ─ except CircuitOpen: bound_log.warning("circuit.open", attempt=attempt) await send_to_dlq(job_id, school_id, run_id, error="circuit_open") return except PermanentError as e: bound_log.error("sync.permanent_failure", error=str(e)) async with get_db() as db: await SyncJobRepo(db).fail_run(run_id, str(e)) await send_to_dlq(job_id, school_id, run_id, error=str(e), permanent=True) return except TransientError as e: if attempt >= MAX_ATTEMPTS: bound_log.error("sync.max_retries_exceeded", attempts=attempt) async with get_db() as db: await SyncJobRepo(db).fail_run(run_id, str(e)) await send_to_dlq(job_id, school_id, run_id, error=str(e), attempts=attempt) return delay = full_jitter(attempt) bound_log.warning("sync.retry", attempt=attempt, delay_s=round(delay, 2)) await asyncio.sleep(delay)
retry/circuit_breaker.py
# apps/worker/src/retry/circuit_breaker.py import asyncio from enum import auto, Enum class State(Enum): CLOSED = auto(); OPEN = auto(); HALF_OPEN = auto() class CircuitOpen(Exception): pass # In-memory (pakai Redis untuk multi-worker production) _registry: dict[str, "CircuitBreaker"] = {} class CircuitBreaker: def __init__(self, key: str, threshold: int = 5, reset_timeout: int = 30): if key not in _registry: _registry[key] = self self.key = key; self.threshold = threshold self.reset_timeout = reset_timeout self._state = State.CLOSED; self._failures = 0; self._opened_at = None else: # Kembalikan instance yang sudah ada existing = _registry[key] self.__dict__ = existing.__dict__ async def __aenter__(self): if self._state == State.OPEN: if asyncio.get_event_loop().time() - self._opened_at > self.reset_timeout: self._state = State.HALF_OPEN else: raise CircuitOpen(f"Circuit {self.key} is OPEN") return self async def __aexit__(self, exc_type, exc, tb): if exc_type is None: # success — reset self._failures = 0; self._state = State.CLOSED else: self._failures += 1 if self._failures >= self.threshold: self._state = State.OPEN self._opened_at = asyncio.get_event_loop().time() return False # jangan suppress exception

# src/db/repositories/sync_job_repo.py from sqlalchemy.ext.asyncio import AsyncSession from sqlalchemy import text, select, update from models.sync_job import SyncJobRun, SyncJobDLQ from uuid import UUID from datetime import datetime, timezone class SyncJobRepo: def __init__(self, session: AsyncSession): self.db = session async def create_run(self, job_id: UUID, school_id: UUID, triggered_by: str = "scheduler") -> str: run = SyncJobRun(job_id=job_id, school_id=school_id, triggered_by=triggered_by) self.db.add(run) await self.db.flush() return str(run.id) async def fetch_pending(self, limit: int = 10) -> list[SyncJobRun]: """FOR UPDATE SKIP LOCKED — aman untuk multi-worker.""" result = await self.db.execute( text(""" SELECT * FROM sync_job_run WHERE status = 'queued' ORDER BY created_at LIMIT :limit FOR UPDATE SKIP LOCKED """), {"limit": limit} ) return result.mappings().all() async def complete_run(self, run_id: str, records_synced: int, records_failed: int = 0): await self.db.execute( update(SyncJobRun) .where(SyncJobRun.id == UUID(run_id)) .values( status="done", records_synced=records_synced, records_failed=records_failed, finished_at=datetime.now(timezone.utc), ) ) await self.db.commit() async def fail_run(self, run_id: str, error_message: str): await self.db.execute( update(SyncJobRun) .where(SyncJobRun.id == UUID(run_id)) .values(status="failed", error_message=error_message[:2048], finished_at=datetime.now(timezone.utc)) ) await self.db.commit() async def upsert_records(self, school_id: UUID, data: list[dict]) -> object: """Bulk upsert dengan ON CONFLICT DO UPDATE — idempotent.""" if not data: return type("R", (), {"inserted": 0, "updated": 0, "failed": 0})() # Gunakan execute_many dengan ON CONFLICT result = await self.db.execute( text(""" INSERT INTO siswa (school_id, nis, nama, kelas, data) VALUES (:school_id, :nis, :nama, :kelas, :data) ON CONFLICT (school_id, nis) DO UPDATE SET nama = EXCLUDED.nama, kelas = EXCLUDED.kelas, data = EXCLUDED.data, updated_at = now() RETURNING (xmax = 0) AS inserted """), [{"school_id": str(school_id), **row} for row in data] ) rows = result.fetchall() inserted = sum(1 for r in rows if r.inserted) return type("R", (), {"inserted": inserted, "updated": len(rows)-inserted, "failed": 0})()

# src/config/otel.py from opentelemetry import trace, metrics from opentelemetry.sdk.trace import TracerProvider from opentelemetry.sdk.trace.sampling import TraceIdRatioBased from opentelemetry.sdk.trace.export import BatchSpanProcessor from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter from opentelemetry.sdk.metrics import MeterProvider from opentelemetry.exporter.prometheus import PrometheusMetricReader from prometheus_client import Counter, Histogram, Gauge def setup_otel(service_name: str): # Traces — 10% sampling, 100% untuk error/slow provider = TracerProvider(sampler=TraceIdRatioBased(0.10)) provider.add_span_processor( BatchSpanProcessor(OTLPSpanExporter(endpoint="otel-collector:4317")) ) trace.set_tracer_provider(provider) # Metrics — expose via /metrics untuk Prometheus scrape metrics.set_meter_provider( MeterProvider(metric_readers=[PrometheusMetricReader()]) ) # Prometheus metrics — didefinisikan sekali di module level api_duration = Histogram( 'http_request_duration_seconds', 'API latency', ['method', 'endpoint', 'status'], buckets=[.01, .05, .1, .2, .3, .5, 1, 2, 5] ) sync_success = Counter('sync_job_success_total', 'Sync berhasil', ['provider']) sync_failure = Counter('sync_job_failure_total', 'Sync gagal', ['provider']) dlq_total = Counter('sync_job_dlq_total', 'Dead letter count', ['reason']) queue_depth = Gauge ('rabbitmq_queue_depth', 'Queue depth', ['queue']) db_conns = Gauge ('db_connections_active', 'DB connections')
